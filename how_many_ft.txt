
/*
Here's the first-touch query, in case you need it
*/
WITH first_touch AS (
    SELECT user_id,
        MIN(timestamp) as first_touch_at
    FROM page_visits
    GROUP BY user_id),
ft_count AS (SELECT ft.user_id,
        ft.first_touch_at,
        pv.utm_source,
        pv.utm_campaign
FROM first_touch ft
JOIN page_visits pv
    ON ft.user_id = pv.user_id
    AND ft.first_touch_at = pv.timestamp)
SELECT ft_count.utm_source, 
       ft_count.utm_campaign,
       count(*)
FROM ft_count
GROUP BY 1,2
ORDER BY 3 DESC;
